```javascript
/**
 * 防抖
 */
function debounce(fn,delay){
    let timer = null;
    return function() {
        if ($element == null){ //如果当时没有传入element对象，从触发事件的对象中取
            console.debug("[url-check] 开始获取当前事件对象",this);
            $element = $(this)
        }
        if(timer){
            clearTimeout(timer)
            timer = setTimeout(fn,delay)
        }else{
            timer = setTimeout(fn,delay)
        }
    }
}
```
